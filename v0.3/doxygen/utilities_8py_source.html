40 def bashify_string(string):
41  """Given a Python string, express it as a bash string.
42 
43  Expresses a python string as a bash string.
44 
45  Example:
46 
47  @code
48  print bashify_string("123''$$")
49  # --> prints '123'"''"'$$'
50  @endcode
51 
52  @param string Any subclass of str
53 
54  @returns valid bash code to represent the string"""
55  output=io.StringIO()
56  for m in re.finditer('''(?xs)
57  (
58  (?P<quotes>'+)
59  | (?P<dquotes>"+)
60  | (?P<printable>[!-&(-\[\]-~ ]+)
61  | (?P<control>.)
62  )''',string):
63  if m.group('quotes'):
64  output.write('"' + m.group('quotes') + '"')
65  elif m.group('dquotes'):
66  output.write("'" + m.group('dquotes') + "'")
67  elif m.group('printable'):
68  output.write("'"+m.group('printable')+"'")
69  elif m.group('control'):
70  output.write("$'\%03o'"%ord(m.group('control')))
71  ret=output.getvalue()
72  output.close()
73  return ret

75 def elipses(long_string,max_length=40,elipses='...'):
76  """!Returns a shortened version of long_string.
77 
78  If long_string is longer than max_length characters, returns a
79  string of length max_length that starts with long_string and ends
80  with elipses. Hence, the number of characters of long_string that
81  will be used is max_length-len(elipses)
82 
83  @param long_string a str or subclass thereof
84  @param max_length maximum length string to return
85  @param elipses the elipses string to append to the end"""
86  strlen=len(long_string)
87  if strlen<max_length:
88  return long_string
89  else:
90  return long_string[0:max_length-len(elipses)]+elipses

104 def dqstring2bracestring(dq):
105  """!Converts a bash-style double quote string to a tripple brace
106  string.
107  @param dq The bash-style double quote string, minus the
108  surrounding double quotes."""
109  output=io.StringIO()
110  for m in re.finditer(r'''(?xs)
111  (
112  \\ (?P<backslashed>.)
113  | (?P<braces> [\]\[]+ )
114  | (?P<text> [^\\@\]\[]+)
115  | (?P<atblock>
116  @ \[ @ \]
117  | @ \[ ' [^']+ ' \]
118  | @ \[ [^\]]+ \]
119  )
120  | (?P<literal_at> @ (?!\[) )
121  | (?P<error> . )
122  ) ''',dq):
123  if m.group('backslashed'):
124  s=m.group('backslashed')
125  if s=='@':
126  output.write('@[@]')
127  elif s in '[]':
128  output.write("@['"+s+"']")
129  else:
130  output.write(s)
131  elif m.group('literal_at'):
132  output.write('@[@]')
133  elif m.group('atblock'):
134  output.write(m.group('atblock'))
135  elif m.group('braces'):
136  output.write("@['"+m.group('braces')+"']")
137  elif m.group('text'):
138  output.write(m.group('text'))
139  else:
140  raise ValueError(
141  'Cannot convert double-quote string \"%s\" to brace string: '
142  'parser error around character \"%s\"."'%(dq,m.group()))
143  value=output.getvalue()
