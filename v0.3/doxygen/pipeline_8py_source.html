70 def call_fcntrl(stream,on,off,logger=None):
71  """!Internal function that implements unblock()
72  @param stream the stream to modify
73  @param on flags to turn on
74  @param off flags to turn off
75  @param logger a logging.Logger for messages

333

335  modifying the implementation of Pipeline. Use the produtil.run
336  module instead of calling launch() and manage().

339  @param inf the input file

345  @param logger Logs to the specified object, at level DEBUG, if a logger is

388

394  errio=io.BytesIO()
395  else:
396  errio=io.StringIO()

398  haveio=True

401  if logger is not None:
402  logger.debug("Monitor process %d."%proc)
403  work.append([2,proc])
404

406  lastproc=time.time()
407  forceclose=False

437  nin+=n

440  logger.debug("Done writing all %d characters; close %d."
441  %(nin,tgt))

443  work.pop(i)
444  continue # do not increment i
445  if forceclose:
446  if logger is not None:
447  logger.debug("Force close of in %d due to timeout."

451  continue # do not increment i

463  else:
464  raise

466  # end of file
467  if logger is not None:
468  logger.debug("eof reading output %d"%tgt)
469  pclose(tgt)

527  # No data in past two seconds and all processes have
528  # exited.
529  forceclose=True
530  continue
531  if work:
532  if logger is not None:
533  logger.debug("Bottom of loop with work=%s"%repr(work))
534  if sleeptime:
535  time.sleep(sleeptime)
536  elif haveio:
537  time.sleep(0.01)
538  else:
539  time.sleep(0.2)
540  if logger is not None:
541  logger.debug("Done monitoring pipeline.")
542 
543  outstr=None
544  if outf is not None:
545  outstr=outio.getvalue()
546  outio.close()
547 
548  errstr=None
549  if errf is not None:
550  errstr=errio.getvalue()
551  errio.close()
552 
553  if _kill_all is not None:
554  raise NoMoreProcesses(
555  "Master thread caught a signal. This thread should exit.")
556  return (outstr, errstr, done)
