240  def format_thresh(self, thresh):
241  """! Format thresholds for file naming
242 
243  Args:
244  thresh - string of the treshold(s)
245 
246  Return:
247  thresh_symbol - string of the threshold(s)

264  if thresh_opt in ['>', 'gt']:
265  thresh_symbol+='>'+thresh_value
266  thresh_letter+='gt'+thresh_value
267  elif thresh_opt in ['>=', 'ge']:
268  thresh_symbol+='>='+thresh_value
269  thresh_letter+='ge'+thresh_value
270  elif thresh_opt in ['<', 'lt']:
271  thresh_symbol+='<'+thresh_value
272  thresh_letter+='lt'+thresh_value
273  elif thresh_opt in ['<=', 'le']:
274  thresh_symbol+='<='+thresh_value
275  thresh_letter+='le'+thresh_value
276  elif thresh_opt in ['==', 'eq']:
277  thresh_symbol+='=='+thresh_value
278  thresh_letter+='eq'+thresh_value
279  elif thresh_opt in ['!=', 'ne']:
280  thresh_symbol+='!='+thresh_value
281  thresh_letter+='ne'+thresh_value
282  return thresh_symbol, thresh_letter
