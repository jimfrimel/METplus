436 def confwalker(conf,start,selector,acceptor,recursevar):
437  """!walks through a ConfigParser-like object performing some action
438 
439  Recurses through a ConfigParser-like object "conf" starting at
440  section "start", performing a specified action. The special
441  variable whose name is in recursevar specifies a list of
442  additional sections to recurse into. No section will be processed
443  more than once, and sections are processed in breadth-first order.
444  For each variable seen in each section (including recursevar),
445  this will call selector(sectionname, varname) to see if the
446  variable should be processed. If selector returns True, then
447  acceptor(section, varname, value) will be called.
448 
449  @param conf the ConfigParser-like object
450  @param start the starting section
451  @param selector a function selector(section,option) that decides
452  if an option needs processing (True) or not (False)
453  @param acceptor a function acceptor(section,option,value)
454  run on all options for which the selector returns True
455  @param recursevar an option in each section that lists more
456  sections the confwalker should touch. If the selector returns
457  True for the recursevar, then the recursevar will be sent to
458  the acceptor. However, it will be scanned for sections to
459  recurse into even if the selector rejects it."""
460  touched=set()
461  requested=[str(start)]
462  while len(requested)>0:
463  sec=requested.pop(0)
464  if sec in touched:
465  continue
466  touched.add(sec)
467  for (key,val) in conf.items(sec):
468  if selector(sec,key):
469  acceptor(sec,key,val)
470  if key==recursevar:
471  for sec2 in reversed(val.split(',')):
472  trim=sec2.strip()
473  if len(trim)>0 and not trim in touched:
474  requested.append(trim)

1123  def _interp(self,sec,opt,morevars=None,taskvars=None):
1124  """!implementation of data-getting routines
1125 
1126  This is the underlying implementation of the various self.get*
1127  routines, and lies below the self._get. It reads a config
1128  option opt from the config section sec.
1129 
1130  If the string contains a {...} expansion, the _interp will
1131  perform string interpolation, expanding {...} strings
1132  according to ConfigParser rules. If the section contains an
1133  @inc, and any variables requested are not found, then the
1134  sections listed in @inc are searched. Failing that, the
1135  config and dir sections are searched.
1136 
1137  @param sec the section name
1138  @param opt the option name
1139  @param morevars a dict containing variables whose values will
1140  override anything in this ProdConfig when performing string
1141  interpolation.
1142  @param taskvars serves the same purpose as morevars, but
1143  provides a second scope.
1144  @return the result of the string expansion"""
1145  sections=( sec, 'config','dir', '@inc' )
1146  gotted=False
1147  for section in sections:
1148  if section=='@inc':
1149  try:
1150  inc=self._conf.get(sec,'@inc')
1151  except NoOptionError:
1152  inc=''
1153  if inc:
1154  touched=set(( sec,'config','dir' ))
1155  for incsection in inc.split(","):
1156  trim=incsection.strip()
1157  if len(trim)>0 and trim not in touched:
1158  touched.add(trim)
1159  try:
1160  got=self._conf.get(trim,opt,raw=True)
1161  gotted=True
1162  break
1163  except (KeyError,NoSectionError,NoOptionError) as e:
1164  pass # var not in section; search elsewhere
1165  if gotted: break
1166  else:
1167  try:
1168  got=self._conf.get(section,opt,raw=True)
1169  gotted=True
1170  break
1171  except (KeyError,NoSectionError,NoOptionError) as e:
1172  pass # var not in section; search elsewhere
1173 
1174  if not gotted:
1175  raise NoOptionError(opt,sec)
1176 
1177  if morevars is None:
1178  return self._formatter.format(got,
1179  __section=sec,__key=opt,__depth=0,__conf=self._conf,
1180  ENV=ENVIRONMENT, __taskvars=taskvars)
1181  else:
1182  return self._formatter.format(got,
1183  __section=sec,__key=opt,__depth=0,__conf=self._conf,
1184  ENV=ENVIRONMENT,__taskvars=taskvars,**morevars)
