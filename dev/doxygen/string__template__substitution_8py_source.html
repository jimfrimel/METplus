60  """ Replace in 'text' all occurrences of any key in the

103 def format_one_time_item(item, time_str, unit):
104  """!Determine precision of time offset value and format
105  Args:
106  @param item format to substitute, i.e. 3M or H
107  @param time_str time value that precision will be applied, i.e. 60
108  @param unit currently being processed, i.e. M or H or S
109  Returns: Padded value or empty string if unit is not found in item
110  """
111  count = item.count(unit)
112  if count > 0:
113  rest = ''
114  # get precision from number (%3H)
115  res = re.match(r"^\.*(\d+)"+unit+"(.*)", item)
116  if res:
117  padding = int(res.group(1))
118  rest = res.group(2)
119  else:
120  padding = count
121  res = re.match("^"+unit+"+(.*)", item)
122  if res:
123  rest = res.group(1)
124  if unit != 's':
125  padding = max(2, count)
126 
127  # add formatted time
128  return str(time_str).zfill(padding)+rest
129 
130  # return empty string if no match
131  return ''

133 def format_hms(fmt, obj):
134  """!For time offset values, get hour, minute, and second values
135  to format as necessary
136  Args:
137  @param fmt format to substitute, i.e. %3H or %2M or %S
138  @param obj time value in seconds to format, i.e. 3600
139  Returns: Formatted time value
140  """
141  out_str = ''
142  fmt_split = fmt.split('%')[1:]
143  # split time into days, hours, mins, and secs
144  # time should be relative to the next highest unit if higher units are specified
145  # i.e. 90 minutes %M => 90, but %H%M => 0130
146  days = obj // 86400
147  hours = obj // 3600
148  minutes = obj // 60
149  seconds = obj
150 
151  # if days are specified, change hours, minutes, and seconds to relative
152  if True in ['d' in x for x in fmt_split]:
153  hours = (obj % 86400) // 3600
154  minutes = (obj % 3600) // 60
155  seconds = (obj % 3600) % 60
156 
157  # if hours are specified, change minutes and seconds to relative
158  if True in ['H' in x for x in fmt_split]:
159  minutes = (obj % 3600) // 60
160  seconds = (obj % 3600) % 60
161 
162  # if minutes are specified, change seconds to relative
163  if True in ['M' in x for x in fmt_split]:
164  seconds = (obj % 3600) % 60
165 
166  # parse format
167  for item in fmt_split:
168  out_str += format_one_time_item(item, hours, 'H')
169  out_str += format_one_time_item(item, minutes, 'M')
170  out_str += format_one_time_item(item, seconds, 'S')
171  out_str += format_one_time_item(item, obj, 's')
172  out_str += format_one_time_item(item, days, 'd')
173 
174  return out_str

332  def do_string_sub(self):
333 
334  """Populates the specified template with information from the
335  kwargs dictionary. The template structure is comprised of
336  a fixed string populated with template place-holders inside curly
337  braces, for example {tmpl_str}. The tmpl_str must be present as
338  a key in the kwargs dictionary, and the value will replace the
339  {tmpl_str} in the returned string.
340 
341  In some cases, the template keys can have parameters containing
342  formatting information. The format of the template in this case
343  is {tmpl_str?parm=val}. The supported parameters are:
344 
345  init, valid:
346  fmt - specifies a strftime format for the date/time
347  e.g. %Y%m%d%H%M%S, %Y%m%d%H
348 
349  lead, level:
350  fmt - specifies an amount of time in [H]HH[MMSS] format
351  e.g. %HH, %HHH, %HH%MMSS, %HHH%MMSS
352 
353  cycle, negative_offset, positive_offset:
354  fmt - specifies the cycle and offset hours in HH format. H and
355  HHH format are supported, to anticipate any changes
356  in prepbufr data.
357 
358  The following were created to support processing of
359  tropical cyclone data:
360  cyclone:
361  fmt - specifies the annual cyclone number as a string.
362 
363  region:
364  fmt - a string that specifies the region/basin of cyclone.
365  For ATCF_by_pairs formatted data, this is a 2-character
366  designation:
367  AL|WP|CP|EP|SH|IO|LS lower case designations are
368  observed in filenames.
369  date:
370  fmt - a string representation of the date format for a
371  subdirectory in which track data resides, or the
372  string representation of the date format in a file.
373  Expected formats: YYYY, YYYYMM, YYYYMMDD, and YYYYMMDDhh
374  misc:
375  fmt -a string that represents any other miscellaneous feature
376  of the track data, such as experiment name or some other descriptor
377 
378 
379  """
380 
381  # The . matches any single character except newline, and the
382  # following + matches 1 or more occurrence of preceding expression.
383  # The ? after the .+ makes it a lazy match so that it stops
384  # after the first "}" instead of continuing to match as many
385  # characters as possible
386 
387  # findall searches through the string and finds all non-overlapping
388  # matches and returns the group
389  # match_list is a list with the contents being the data between the
390  # curly braces
391  match_list = re.findall(r'\{(.+?)\}', self.tmpl)
392 
393  # finditer gets the start and end indices
394  # Iterate over each match to get the starting and ending indices
395  matches = re.finditer(r'\{(.+?)\}', self.tmpl)
396  match_start_end_list = []
397  for match in matches:
398  match_start_end_list.append((match.start(), match.end()))
399 
400  if len(match_list) == 0:
401  return self.tmpl
402 
403  if len(match_list) != len(match_start_end_list):
404  # Log and exit
405  self.logger.error("match_list and match_start_end_list should " +
406  "have the same length for template: " +
407  self.tmpl)
408  exit(1)
409 
410  # A dictionary that will contain the string to replace (key)
411  # and the string to replace it with (value)
412  replacement_dict = {}
413 
414  # Search for the FORMATTING_DELIMITER within the first string
415  for match in match_list:
416 
417  string_to_replace = TEMPLATE_IDENTIFIER_BEGIN + match + \
418  TEMPLATE_IDENTIFIER_END
419  split_string = match.split(FORMATTING_DELIMITER)
420 
421  # valid, init, lead, etc.
422  # print split_string[0]
423  # value e.g. 2016012606, 3
424 
425  # split_string[0] holds the key (e.g. "init", "valid", etc)
426  if split_string[0] not in self.kwargs.keys():
427  # Log and exit
428  self.logger.error("The key " + split_string[0] +
429  " was not passed to StringSub " +
430  " for template: " + self.tmpl)
431  exit(1)
432 
433  # if shift is set, get that value before handling formatting
434  for split_item in split_string:
435  if split_item.startswith(SHIFT_STRING):
436  self.shift_seconds = get_seconds_from_template(split_item)
437 
438  # if truncate is set, get that value before handling formatting
439  for split_item in split_string:
440  if split_item.startswith(TRUNCATE_STRING):
441  self.truncate_seconds = get_seconds_from_template(split_item)
442 
443  # format times appropriately and add to replacement_dict
444  formatted = False
445  for idx, split_item in enumerate(split_string):
446  if split_item.startswith(FORMAT_STRING):
447  replacement_dict[string_to_replace] = \
448  self.handle_format_delimiter(split_string, idx)
449  formatted = True
450 
451  # No formatting or length is requested
452  if not formatted:
453  # Add back the template identifiers to the matched
454  # string to replace and add the key, value pair to the
455  # dictionary
456  value = self.kwargs.get(split_string[0], None)
457  if isinstance(value, int):
458  value = f"{value}S"
459  replacement_dict[string_to_replace] = value
460 
461 
462  # reset shift seconds so it doesn't apply to next match
463  self.shift_seconds = 0
464  self.truncate_seconds = 0
465 
466  # Replace regex with properly formatted information
467  self.tmpl = multiple_replace(replacement_dict, self.tmpl)
468  return self.tmpl

478  def get_fmt_info(self, fmt, full_str, match_dict, identifier):
479  """!Reads format information from tag and populates dictionary with
480  extracted values.
481  Args:
482  @param fmt formatting values from template tag, i.e. %Y%m%d
483  @param full_str rest of text from filename that can be parsed
484  @param match_dict dictionary of extracted information. Key is made up
485  of the identifier and the format tag, i.e. init_H or valid_M.
486  Value is the extracted information, i.e. 19870201
487  @param identifier tag name, i.e. 'init' or 'lead'
488  Returns: Number of characters processed from the filename if success,
489  -1 if failed to parse all format items in template tag"""
490  length = 0
491  match_list = re.findall('%[^%]+', fmt)
492  for match in match_list:
493  new_len = 0
494  extra_len = 0
495 
496  # exact match, i.e. %Y
497  if match in LENGTH_DICT.keys():
498  # handle lead and level that have 1 digit precision
499  new_len = LENGTH_DICT[match]
500  if match == '%H' and identifier == 'lead' or \
501  identifier == 'level':
502  # look forward until non-digit is found
503  new_len = 0
504  while full_str[new_len].isdigit():
505  new_len += 1
506 
507  length += new_len
508  if not add_to_dict(identifier+'+'+match[1:], match_dict,
509  full_str, new_len):
510  return -1
511  # if match starts with key, find new length, i.e. %Y: or %HH
512  elif match[0:2] in LENGTH_DICT.keys():
513  match_char = match[1]
514  match_len = re.match('%(['+match_char+']+)(.*)', match)
515  if match_len:
516  new_len = len(match_len.group(1))
517  if new_len > 1:
518  length += new_len
519  else:
520  match_begin = match[0:2]
521  new_len = LENGTH_DICT[match_begin]
522  length += new_len
523  if not add_to_dict(identifier+'+'+match_char, match_dict,
524  full_str, new_len):
525  return -1
526 
527  if len(match_len.group(2)) > 0:
528  extra_len = len(match_len.group(2))
529  length += extra_len
530  else:
531  # match %2H or %.2H
532  match_len = re.match(r'%\.*(\d+)(\D)$', match)
533  if match_len:
534  new_len = int(match_len.group(1))
535  length += new_len
536  if not add_to_dict(identifier+'+'+match_len.group(2)[0],
537  match_dict, full_str, new_len):
538  return -1
539  else:
540  return -1
541 
542  full_str = full_str[new_len+extra_len:]
543 
544  return length
