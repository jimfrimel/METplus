5 # This module contains classes and functions to assist developers in

27 def guess_total_tasks_impl(logger,silent):
28  total_tasks=os.environ.get('TOTAL_TASKS','')
29  if total_tasks: return int(total_tasks,0)
30  pbs_np=os.environ.get('PBS_NP','')
31  pbs_num_ppn=os.environ.get('PBS_NUM_PPN','')
32  if not pbs_np:
33  raise KeyError('TOTAL_TASKS')
34  np=int(pbs_np,10)
35  if not pbs_num_ppn:
36  return np
37  ppn=int(pbs_num_ppn,10)
38  if ppn<2:
39  # workaround for Theia/Jet issue. Not ideal
40  cpus=set()
41  cpu_cores=None
42  with open('/proc/cpuinfo','rt') as cpuinfo:
43  for line in cpuinfo:
44  if line.find('physical id')>=0:
45  cpus.add(line.strip())
46  m=re.search(r'cpu cores\s+:\s+(\d+)',line)
47  if m:
48  cpu_cores=int(m.group(1),10)
49  if not cpus or not cpu_cores:
50  raise KeyError('TOTAL_TASKS')
51  ppn=len(cpus)*cpu_cores
52  if logger and not silent:
53  logger.info('%d cpus with %d physical cores each = %d physical cores per node'%(
54  len(cpus),cpu_cores,ppn))
55  return np*ppn

177 class CMDFGen(object):
178  """!Generates files with one line per MPI rank, telling what
179  program to run on each rank.
180 
181  This class is used to generate command files for mpiserial, poe or
182  mpirun.lsf. Command files are files with one MPI rank per line
183  containing a shell command to run for that rank. Generally the
184  input (lines) is generated by the to_arglist function in a
185  subclass of produtil.mpiprog.MPIRanksBase. See the
186  produtil.mpi_impl.mpirun_lsf for an example of how to use this."""
187  def __init__(self,base,lines,cmd_envar='SCR_CMDFILE',
188  model_envar=None,filename_arg=False,
189  silent=False,filename_option=None,
190  next_prerun=None,**kwargs):
191  """!CMDFGen constructor
192 
193  @param base type of command file being generated. See below.
194  @param lines the command file contents as a list of strings, one per line
195  @param cmd_envar environment variable to set to the command file path
196  @param model_envar environment variable to set to "MPMD"
197  @param kwargs Sets the command file name. See below.
198  @param filename_arg If True, the name of the command file is appended to the program argument list.
199  @param filename_option A string or None. If filename_arg is true, this string is appended before the filename_arg
200 
201  The command file is generated from
202  tempfile.NamedTemporaryFile, passing several arguments from
203  kwargs, if provided, or suitable defaults otherwise. There
204  are several arguments used. In all cases, replace "base" with
205  the contents of the @c base argument:
206 
207  * base_suffix --- temporary file suffix (default: "base.")
208  * base_prefix --- temporary file prefix (default: ".cmdf")
209  * base_tempdir --- directory in which to create the file
210 
211  @bug The base_suffix keyword is used for both the suffix and prefix"""
212  assert(base is not None)
213  assert(isinstance(lines,list))
214  assert(len(lines)>0)
215  assert(isinstance(lines[0],str))
216  assert(len(lines[0])>0)
217  self.filename=kwargs.get(str(base),None)
218  self.tmpprefix=kwargs.get('%s_suffix'%(base,),'%s.'%(base,))
219  self.tmpsuffix=kwargs.get('%s_suffix'%(base,),'.cmdf')
220  self.tmpdir=kwargs.get('%s_tmpdir'%(base,),'.')
221  self.cmd_envar=cmd_envar
222  self.model_envar=model_envar
223  self.filename_arg=filename_arg
224  self.filename_option=filename_option
225  self.silent=bool(silent)
226  self.next_prerun=next_prerun
227  out='\n'.join(lines)
228  if len(out)>0:
229  out+='\n'
230  self.cmdf_contents=out
231  return
