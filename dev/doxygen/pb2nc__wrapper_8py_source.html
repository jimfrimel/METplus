152  def reformat_grid_id(self, grid_id):
153  """!Reformat the grid id (MASK_GRID value in the configuration
154  file) if it starts with G. Looks for G<n> where n is a
155  digit 0-999 and zero pads the value. i.e. G7 becomes G007,
156  G13 becomes G013, etc.
157  Args:
158  @param grid_id identifier of grid
159  Returns:
160  @return reformatted grid id if valid, None if not
161  """
162  # If grid ID does not start with G, return it
163  if not grid_id.startswith('G'):
164  return grid_id
165 
166  # look for G<n> where n is a digit 0-999
167  match = re.match(r'G([0-9]{1,3})$', grid_id)
168 
169  # pad with zeros if found
170  if match:
171  number = match.group(1)
172  return 'G' + number.zfill(3)
173 
174  # Unexpected format
175  self.log_error('Grid id in unexpected format of Gn or ' +
176  'Gnn, please check again. Exiting...')
177  return None
