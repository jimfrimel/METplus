381

384 
385  # check if there is a list of files provided in the template
386  # process each template in the list (or single template)
387  template_list = [template.strip() for template in input_template.split(',')]
388 
389  # return None if a list is provided for a wrapper that doesn't allow
390  # multiple files to be processed
391  if len(template_list) > 1 and not self.c_dict['ALLOW_MULTIPLE_FILES']:

394  return None

397  # perform string substitution
398  dsts = sts.StringSub(self.logger,

403

406 
407  self.logger.debug(f"Looking for {data_type} file {full_path}")
408 
409  # if wildcard expression, get all files that match

411  if not self.c_dict['ALLOW_MULTIPLE_FILES']:
412  self.log_error("Wildcard character found in file path when using wrapper that "

414  return None

417  self.logger.debug(f'Wildcard file pattern: {full_path}')
418  self.logger.debug(f'{str(len(wildcard_files))} files match pattern')
419 
420  # add files to list of files
421  for wildcard_file in wildcard_files:
422  check_file_list.append(wildcard_file)
423  else:
424  # add single file to list
425  check_file_list.append(full_path)

451  def find_file_in_window(self, level, data_type, time_info, mandatory=True, return_list=False):
452  template = self.c_dict[f'{data_type}_INPUT_TEMPLATE']
453  data_dir = self.c_dict[f'{data_type}_INPUT_DIR']
454 
455  # convert valid_time to unix time
456  valid_time = time_info['valid_fmt']
457  valid_seconds = int(datetime.strptime(valid_time, "%Y%m%d%H%M%S").strftime("%s"))
458  # get time of each file, compare to valid time, save best within range
459  closest_files = []
460  closest_time = 9999999
461 
462  # get range of times that will be considered
463  valid_range_lower = self.c_dict[data_type + '_FILE_WINDOW_BEGIN']
464  valid_range_upper = self.c_dict[data_type + '_FILE_WINDOW_END']
465  lower_limit = int(datetime.strptime(util.shift_time_seconds(valid_time, valid_range_lower),
466  "%Y%m%d%H%M%S").strftime("%s"))
467  upper_limit = int(datetime.strptime(util.shift_time_seconds(valid_time, valid_range_upper),
468  "%Y%m%d%H%M%S").strftime("%s"))
469 
470  msg = f"Looking for {data_type} files under {data_dir} within range " +\
471  f"[{valid_range_lower},{valid_range_upper}] using template {template}"
472  self.logger.debug(msg)
473 
474  if not data_dir:
475  self.log_error('Must set INPUT_DIR if looking for files within a time window')
476  return None
477 
478  # step through all files under input directory in sorted order
479  for dirpath, _, all_files in os.walk(data_dir):
480  for filename in sorted(all_files):
481  fullpath = os.path.join(dirpath, filename)
482 
483  # remove input data directory to get relative path
484  rel_path = fullpath.replace(f'{data_dir}/', "")
485  # extract time information from relative path using template
486  file_time_info = util.get_time_from_file(self.logger, rel_path, template)
487  if file_time_info is not None:
488  # get valid time and check if it is within the time range
489  file_valid_time = file_time_info['valid'].strftime("%Y%m%d%H%M%S")
490  # skip if could not extract valid time
491  if file_valid_time == '':
492  continue
493  file_valid_dt = datetime.strptime(file_valid_time, "%Y%m%d%H%M%S")
494  file_valid_seconds = int(file_valid_dt.strftime("%s"))
495  # skip if outside time range
496  if file_valid_seconds < lower_limit or file_valid_seconds > upper_limit:
497  continue
498 
499  # if only 1 file is allowed, check if file is
500  # closer to desired valid time than previous match
501  if not self.c_dict['ALLOW_MULTIPLE_FILES']:
502  diff = abs(valid_seconds - file_valid_seconds)
503  if diff < closest_time:
504  closest_time = diff
505  del closest_files[:]
506  closest_files.append(fullpath)
507  # if multiple files are allowed, get all files within range
508  else:
509  closest_files.append(fullpath)
510 
511  if not closest_files:
512  msg = f"Could not find {data_type} files under {data_dir} within range " +\
513  f"[{valid_range_lower},{valid_range_upper}] using template {template}"
514  if mandatory:
515  self.log_error(msg)
516  else:
517  self.logger.warning(msg)
518  return None
519 
520  # check if file(s) needs to be preprocessed before returning the path
521  # if one file was found and return_list if False, return single file
522  if len(closest_files) == 1 and not return_list:
523  return util.preprocess_file(closest_files[0],
524  self.c_dict[data_type + '_INPUT_DATATYPE'],
525  self.config)
526 
527  # return list if multiple files are found
528  out = []
529  for close_file in closest_files:
530  outfile = util.preprocess_file(close_file,
531  self.c_dict[data_type + '_INPUT_DATATYPE'],
532  self.config)
533  out.append(outfile)
534 
535  return out
