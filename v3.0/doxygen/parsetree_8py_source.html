1730 def pack_ranks(nodesize,count):
1731  """!Spreads count MPI ranks over the fewest possible nodes whose
1732  maximum PETs per node (ppn) is nodesize. Attempts to distribute
1733  the ranks per node as equally as possible. Any node will never
1734  have more than one or less than one rank than any other node.
1735 
1736  @returns a list of two-element lists. Each two element list
1737  contains a pair [number of nodes, PETs per node (ppn)] for nodes
1738  that have an identical number of PETs per node (ppn)
1739  @param nodesize the maximum number of PETS per node of any given compute node
1740  @param count the number of MPI ranks"""
1741  out=list()
1742  n=count
1743  if n<nodesize: # special case: smaller than one node
1744  out.append([1,n])
1745  elif nodesize*(n//nodesize)==n: #exact number of nodes
1746  out.append([n//nodesize,nodesize])
1747  else:
1748  need=math.ceil(n/float(nodesize))
1749  averagef=n/math.ceil(need)
1750  af,ai = math.modf(averagef)
1751  n1=need-round(af*need)
1752  n2=round(af*need)
1753  if n1: out.append([n1, ai])
1754  if n2: out.append([n2, ai+1])
1755  return out
