221 class NOAAWCOSS(Cluster):
222  """!Represents the NOAA WCOSS clusters, Tide, Gyre and the test
223  system Eddy.
224 
225  Automatically determines which WCOSS the program is on based on
226  the first letter of socket.gethostname(). Will report no ACL
227  support, and no group quotas. Hence, the cluster should use group
228  IDs for access control.
229

231  is now a property, which may open the /etc/prod file. The result
232  of the self.production property is cached for up to
233  prod_cache_time seconds. That time can be specified in the
234  constructor, and defaults to 30 seconds."""
235  def __init__(self,prod_cache_time=30,name=None,phase=None):
236  """!Creates a NOAAWCOSS object, and optionally specifies the

240  if name is None:

272  @property
273  def wcoss_phase(self):
274  """!Returns integer 1 or 2 for WCOSS Phase 1 or WCOSS Phase 2,
275  respectively. Returns 0 if the request is invalid, such as on
276  WCOSS Cray.
277 
278  Scans /proc/cpuinfo for processor 32. If processor 32 is
279  found, you are on Phase 2, which has 48 virtual processors per
280  node. Otherwise, you are on Phase 1.
281 
282  @note Cached results are returned if available. Use uncache()
283  to force regeneration of the information.
284 
285  @returns 1 for WCOSS Phase 1, 2 for WCOSS Phase 2, or 0 for
286  WCOSS Cray"""
287  if self._phase is None:
288  phase=1
289  with open('/proc/cpuinfo','rt') as cpuinfo:
290  for line in cpuinfo:
291  if re.match(r'(?i)^processor\s*:\s*32',line):
292  phase=2
293  break
294  self._phase=phase
295  return self._phase

297  @property
298  def production(self):
299  """!Is this the WCOSS production machine?
300 
301  The name of the WCOSS production machine: tide, gyre, surge or
302  luna as determined by the /etc/prod file.
303 
304  @returns True or False: is this the WCOSS production machine?
305 
306  @note The return value may change during the execution of this
307  program if a production switch happened. A cached value is
308  returned if the values is not too old. To force a refresh,
309  call uncache() first.
310 
311  @warning The check requires opening and parsing the /etc/prod
312  file, so the runtime is likely several milliseconds when the
313  cache times out."""
314  now=int(time.time())
315  if self._production is None or \
316  now-self._lastprod>self._prod_cache_time:
317  prod=False
318  if os.path.exists('/etc/prod'):
319  with open('/etc/prod','rt') as f:
320  for line in f:
321  if re.match('[a-z]+',line):
322  prod = line.strip()==self.name
323  break
324  self._production=prod
325  self._lastprod=int(time.time())
326  return prod
327  else:
328  return self._production
