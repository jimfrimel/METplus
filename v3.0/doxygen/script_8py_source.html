53 function deliver_file() {
54  local src tgt parent
55  set -e
56  src="$1"
57  tgt="$2"
58  if [[ -d "$src" ]] ; then
59  tgt=$( echo "$tgt" | sed 's,/*$,,g' )
60  rm -rf "$tgt"
61  cp -rp "$src" "$tgt"
62  else
63  # If the tgt is a directory, place the source within.
64  # If the tgt does not exist, but appears to be the name of a
65  # directory, make that directory and place the source within.
66  # /path/to/dirname/ <-- ends with /
67  # /path/to/place/. <-- . is always a directory
68  # /path/to/place/.. <-- .. is always a directory
69  if [[ -d "$tgt" || \
70  ! -e "$tgt" && "${tgt: -1}" == / || \
71  ! -e "$tgt" && "$( basename $tgt )" == . || \
72  ! -e "$tgt" && "$( basename $tgt )" == .. ]] ; then
73  tgt="$tgt"/$( basename "$src" )
74  fi
75  parent=$( dirname "$tgt" )
76  if [[ ! -d "$parent" ]] ; then
77  mkdir -p "$parent"
78  fi
79  tmpfile="$tgt.$$.$RANDOM.$RANDOM"
80  cp -fp "$src" "$tmpfile"
81  mv -fT "$tmpfile" "$tgt"
82  fi
83 }
84 function comparison_wrapper() {
85  local src tgt bn result origtgt
86  set -e
87  set +x
88  cmd="$1"
89  message="$2"
90  src="$3"
91  tgt="$4"

94  tgt="$tgt/$bn"

96  bn=$( basename "$tgt" )
97  src="$src/$bn"
98  fi
99  echo TARGET: $( md5sum $tgt )
100  ls -l $tgt
101  echo SOURCE: $( md5sum $src )
102  ls -l $src
103  set +e
104  if [[ ! -e "$src" ]] ; then
105  report_failure "$tgt: MISSING BASELINE FILE"
106  return 1
107  elif [[ ! -e "$tgt" ]] ; then
108  report_failure "$src: MISSING OUTPUT FILE"
109  return 1
110  fi
111  set -x
112  $cmd "$src" "$tgt"
113  result=$?
114  if [[ "$result" != 0 ]] ; then
115  report_failure "$src: $message MISMATCH"
116  else
117  report_line "$src: $message identical"
118  fi

205  """!Generates the mpi launching command for the given
206  produtil.testing.parsetree.SpawnProcess object
207 
208  @param spawnProcess a description of the process to execute
209  @returns a string containing the mpirun command"""
210  cmd=None
211  is_mpi=False
212  mpiimpl=self.mpiimpl
213  for rank in spawnProcess.iterrank():
214 
215  ranks=rank.ranks(self)
216  ppn=rank.ppn(self)
217  threads=rank.threads(self)
218 
219  if cmd is not None and not ranks:
220  raise PTParserError('Error: mixing MPI and non-MPI programs in the same command is not supported.')
221 
222  args=[ arg.string_context(self) for arg in rank.args ]
223 
224  if ranks:
225  cmdpart=produtil.run.mpi(args[0])
226  cmdpart=cmdpart[args[1:]]
227  is_mpi=True
228  else:
229  cmdpart=produtil.run.exe(args[0],mpiimpl=mpiimpl)
230  cmdpart=cmdpart[args[1:]]
231 
232  if ranks and ppn:
233  cmdpart=cmdpart.rpn(ppn)
234 
235  if threads:
236  cmdpart=produtil.run.openmp(cmdpart,threads=threads,mpiimpl=mpiimpl)
237 
238  if ranks:
239  cmdpart=cmdpart*ranks
240 
241  if cmd is None:
242  cmd=cmdpart
243  else:
244  cmd=cmd+cmdpart
245

247  cmd=produtil.run.mpirun(cmd,mpiimpl=mpiimpl,label_io=True,scheduler_distribution=distribution)
