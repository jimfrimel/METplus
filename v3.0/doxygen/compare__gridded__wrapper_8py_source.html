303  def get_field_info(self, v_name, v_level, v_thresh, v_extra, d_type):
304  """! Format field information into format expected by MET config file
305  Args:
306  @param v_level level of data to extract
307  @param v_thresh threshold value to use in comparison
308  @param v_name name of field to process
309  @param v_extra additional field information to add if available
310  @param d_type type of data to find (FCST or OBS)
311  @rtype string
312  @return Returns formatted field information
313  """
314  # separate character from beginning of numeric level value if applicable
315  _, level = util.split_level(v_level)
316 
317  # list to hold field information
318  fields = []
319 
320  # get cat thresholds if available
321  cat_thresh = ""
322  threshs = [None]
323  if len(v_thresh) != 0:
324  threshs = v_thresh
325  cat_thresh = "cat_thresh=[ " + ','.join(threshs) + " ];"
326 
327  # if neither input is probabilistic, add all cat thresholds to same field info item
328  if not self.c_dict['FCST_IS_PROB'] and not self.c_dict['OBS_IS_PROB']:
329 
330  # if pcp_combine was run, use name_level, (*,*) format
331  # if not, use user defined name/level combination
332  if d_type != 'ENS' and self.config.getbool('config', d_type + '_PCP_COMBINE_RUN', False):
333  field = "{ name=\"" + v_name + "_" + level + \
334  "\"; level=\"(*,*)\";"
335  else:
336  field = "{ name=\"" + v_name + "\";"
337 
338  # add level if it is set
339  if v_level:
340  field += " level=\"" + v_level + "\";"
341 
342  # add threshold if it is set
343  if cat_thresh:
344  field += ' ' + cat_thresh
345 
346  # add extra info if it is set
347  if v_extra:
348  field += ' ' + v_extra
349 
350  field += ' }'
351  fields.append(field)
352 
353  # if either input is probabilistic, create separate item for each threshold
354  else:
355 
356  # if input currently being processed if probabilistic, format accordingly
357  if self.c_dict[d_type + '_IS_PROB']:
358  # if probabilistic data for either fcst or obs, thresholds are required
359  # to be specified or no field items will be created. Create a field dict
360  # item for each threshold value
361  for thresh in threshs:
362  # if utilizing python embedding for prob input, just set the
363  # field name to the call to the script
364  if util.is_python_script(v_name):
365  field = "{ name=\"" + v_name + "\"; prob=TRUE;"
366  elif self.c_dict[d_type + '_INPUT_DATATYPE'] == 'NETCDF' or \
367  not self.c_dict[d_type + '_PROB_IN_GRIB_PDS']:
368  field = "{ name=\"" + v_name + "\";"
369  if v_level:
370  field += " level=\"" + v_level + "\";"
371  field += " prob=TRUE;"
372  else:
373  # a threshold value is required for GRIB prob DICT data
374  if thresh is None:
375  self.log_error('No threshold was specified for probabilistic '
376  'forecast GRIB data')
377  return None
378 
379  thresh_str = ""
380  thresh_tuple_list = util.get_threshold_via_regex(thresh)
381  for comparison, number in thresh_tuple_list:

384  if comparison in ["lt", "le", "<", "<=", "==", "eq"]:
385  thresh_str += "thresh_hi=" + str(number) + "; "
386 
387  field = "{ name=\"PROB\"; level=\"" + v_level + \
388  "\"; prob={ name=\"" + v_name + \
389  "\"; " + thresh_str + "}"
390 
391  # add probabilistic cat thresh if different from default ==0.1
392  prob_cat_thresh = self.c_dict[d_type + '_PROB_THRESH']
393  if prob_cat_thresh is not None:

395 
396  if v_extra:

398 
399  field += ' }'

404  # if pcp_combine was run, use name_level, (*,*) format
