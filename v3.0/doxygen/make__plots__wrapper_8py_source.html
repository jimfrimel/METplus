170  def set_lists_loop_or_group(self, config_lists_to_group_items,
171  config_lists_to_loop_items, config_dict):
172  """! Determine whether the lists from the METplus config file
173  should treat the items in that list as a group or items
174  to be looped over based on user settings, the values
175  in the list, and process being run.
176 
177  Args:
178  config_lists_to_group_items - list of the METplus
179  config list names
180  to group the list's
181  items set by user
182  config_lists_to_loop_items - list of the METplus
183  config list names
184  to loop over the
185  list's items set by
186  user
187  config_dict - dictionary containing
188  the configuration
189  information
190 
191  Returns:
192  lists_to_group_items - list of all the list names
193  whose items are being
194  grouped together
195  lists_to_loop_items - list of all the list names
196  whose items are being
197  looped over
198  """
199  expected_config_lists = [
200  'MODEL_LIST', 'DESC_LIST',
201  'FCST_LEAD_LIST', 'OBS_LEAD_LIST',
202  'FCST_VALID_HOUR_LIST', 'FCST_INIT_HOUR_LIST',
203  'OBS_VALID_HOUR_LIST', 'OBS_INIT_HOUR_LIST',
204  'FCST_VAR_LIST', 'OBS_VAR_LIST',
205  'FCST_UNITS_LIST', 'OBS_UNITS_LIST',
206  'FCST_LEVEL_LIST', 'OBS_LEVEL_LIST',
207  'VX_MASK_LIST', 'INTERP_MTHD_LIST',
208  'INTERP_PNTS_LIST', 'FCST_THRESH_LIST',
209  'OBS_THRESH_LIST', 'COV_THRESH_LIST',
210  'ALPHA_LIST', 'LINE_TYPE_LIST',
211  'STATS_LIST'
212  ]
213  lists_to_group_items = config_lists_to_group_items
214  lists_to_loop_items = config_lists_to_loop_items
215  for config_list in expected_config_lists:
216  if (not config_list in config_lists_to_group_items
217  and not config_list in config_lists_to_loop_items):
218  if config_list == 'LINE_TYPE_LIST' or config_list == 'STATS_LIST':
219  lists_to_group_items.append(config_list)
220  elif config_dict[config_list] == []:
221  self.logger.warning(config_list+" is empty, "
222  +"will be treated as group.")
223  lists_to_group_items.append(config_list)
224  else:
225  lists_to_loop_items.append(config_list)
226  elif (config_list in config_lists_to_loop_items
227  and config_dict[config_list] == []):
228  self.logger.warning(config_list+" is empty, "
229  +"will be treated as group.")
230  lists_to_group_items.append(config_list)
231  lists_to_loop_items.remove(config_list)
232  if (config_list == 'MODEL_LIST'
233  or config_list == 'FCST_LEAD_LIST'
234  or config_list == 'FCST_LEVEL_LIST'
235  or config_list == 'OBS_LEVEL_LIST'
236  or config_list == 'FCST_THRESH_LIST'
237  or config_list == 'OBS_THRESH_LIST'
238  or config_list == 'FCST_UNITS_LIST'
239  or config_list == 'OBS_UNITS_LIST'):
240  if config_list not in lists_to_group_items:
241  lists_to_group_items.append(config_list)
242  if config_list in lists_to_loop_items:
243  lists_to_loop_items.remove(config_list)
244  self.logger.debug("Items in these lists will be grouped together: "
245  +', '.join(lists_to_group_items))
246  self.logger.debug("Items in these lists will be looped over: "
247  +', '.join(lists_to_loop_items))
248  return lists_to_group_items, lists_to_loop_items

250  def parse_model_info(self):
251  """! Parse for model information.
252 
253  Args:
254 
255  Returns:
256  model_list - list of dictionaries containing
257  model information
258  """
259  model_info_list = []
260  all_conf = self.config.keys('config')
261  model_indices = []
262  regex = re.compile(r'MODEL(\d+)$')
263  for conf in all_conf:
264  result = regex.match(conf)
265  if result is not None:
266  model_indices.append(result.group(1))
267  for m in model_indices:
268  if self.config.has_option('config', 'MODEL'+m):
269  model_name = self.config.getstr('config', 'MODEL'+m)
270  model_reference_name = (
271  self.config.getstr('config', 'MODEL'+m+'_REFERENCE_NAME',
272  model_name)
273  )
274  if self.config.has_option('config', 'MODEL'+m+'_OBTYPE'):
275  model_obtype = (
276  self.config.getstr('config', 'MODEL'+m+'_OBTYPE')
277  )
278  else:
279  self.log_error("MODEL"+m+"_OBTYPE was not set.")
280  exit(1)
281  mod = {}
282  mod['name'] = model_name
283  mod['reference_name'] = model_reference_name
284  mod['obtype'] = model_obtype
285  model_info_list.append(mod)
286  return model_info_list, model_indices

542  def create_plots(self, verif_case, verif_type):
543  """! Set up variables and general looping for creating
544  verification plots
545 
546  Args:
547  verif_case - string of the verification case to make
548  plots for
549  verif_type - string of the verification type to make
550  plots for
551 
552  Returns:
553  """
554  # Do checks for bad configuration file options.
555  bad_config_variable_list = [
556  'FCST_VAR_LIST', 'FCST_LEVEL_LIST',
557  'FCST_THRESH_LIST', 'FCST_UNITS_LIST',
558  'OBS_VAR_LIST', 'OBS_LEVEL_LIST',
559  'OBS_THRESH_LIST', 'OBS_UNITS_LIST'
560  ]
561  for bad_config_variable in bad_config_variable_list:
562  if self.config.has_option('config',
563  bad_config_variable):
564  self.log_error("Bad config option for running MakePlots. "
565  "Please remove "+bad_config_variable+" "
566  +"and set using FCST/OBS_VARn")
567  exit(1)
568  loop_group_accepted_options = [
569  'FCST_VALID_HOUR_LIST', 'FCST_INIT_HOUR_LIST',
570  'OBS_VALID_HOUR_LIST', 'OBS_INIT_HOUR_LIST'
571  ]
572  for config_list in self.c_dict['GROUP_LIST_ITEMS']:
573  if config_list not in loop_group_accepted_options:
574  self.log_error("Bad config option for running MakePlots. "
575  +"Only accepted values in GROUP_LIST_ITEMS "
576  +"are FCST_VALID_HOUR_LIST, "
577  +"FCST_INIT_HOUR_LIST, "
578  +"OBS_VALID_HOUR_LIST, "
579  +"OBS_INIT_HOUR_LIST. "
580  +"Found "+config_list)
581  exit(1)
582  for config_list in self.c_dict['LOOP_LIST_ITEMS']:
583  if config_list not in loop_group_accepted_options:
584  self.log_error("Bad config option for running MakePlots. "
585  +"Only accepted values in LOOP_LIST_ITEMS "
586  +"are FCST_VALID_HOUR_LIST, "
587  +"FCST_INIT_HOUR_LIST, "
588  +"OBS_VALID_HOUR_LIST, "
589  +"OBS_INIT_HOUR_LIST. "
590  +"Found "+config_list)
591  exit(1)
592  # Do checks for required configuration file options that are
593  # defined by user.
594  required_config_variable_list = [
595  'VX_MASK_LIST', 'FCST_LEAD_LIST', 'LINE_TYPE_LIST'
596  ]
597  for required_config_variable in required_config_variable_list:
598  if len(self.c_dict[required_config_variable]) == 0:
599  self.log_error(required_config_variable+" has no items. "
600  +"This list must have items to run "
601  +"MakePlots.")
602  exit(1)
603  # Do some preprocessing, formatting, and gathering
604  # of config information.
605  date_type = self.c_dict['DATE_TYPE']
606  formatted_c_dict = copy.deepcopy(self.c_dict)
607  model_info_list, model_indices = self.parse_model_info()
608  if self.c_dict['MODEL_LIST'] == []:
609  if model_indices > 0:
610  self.logger.warning("MODEL_LIST was left blank, "
611  +"creating with MODELn information.")
612  model_name_list = []
613  for model_info in model_info_list:
614  model_name_list.append(model_info['name'])
615  formatted_c_dict['MODEL_LIST'] = model_name_list
616  else:
617  self.log_error("No model information was found.")
618  exit(1)
619  model_obtype_list = []
