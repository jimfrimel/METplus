54 class partial_ordering(object):
55  """!Sorts a pre-determined list of objects, placing unknown items
56  at a specified location.
57 
58  This class is a drop-in replacement for cmp in sorting routines.
59  It represents a partial ordering of objects by specifying the
60  order of a known subset of those objects, and inserting all
61  unknown objects in a specified location in the that list (at the
62  end, by default) in an order determined by cmp(a,b). Example:
63 
64  @code{.py}
65  p=partial_ordering([3,2,1]) # list is ordered as [3,2,1] with
66  # everything else after that
67 
68  sorted([0,1,2,3,6,4,5],p) # = [3, 2, 1, 0, 4, 5, 6]
69  p(1,-99) # = -1, so -99 goes after 1 since -99 is not
70  # in the partial ordering
71  p(1,3) # = 1, so 3 goes before 1 since 3 is before 1
72  # in the partial ordering
73  p(5,10) # = -1 since cmp(5,10)=-1
74  @endcode"""
75  def __init__(self,ordering,unordered=None,backup_keygen=None):
76  """!partial_ordering constructor.
77 
78  Creates a partial ordering. The subset that is ordered is
79  specified by the ordered iterable "ordered" while the index at
80  which to place unordered values is optionally specified by
81  "unordered", which can be anything that can be compared to an
82  int via less than and greater than. If "unordered" is
83  missing, then all objects not in "ordered" will be placed at
84  the end of any list. To place at the beginning of the list,
85  give unordered=0. To insert between the first and second
86  elements, specify 1, between second and third elements:
87  specify unordered=2, and so on. Specify another tiebreaker
88  key generator with "backup_keygen" (default: original value).
89 
90  @param ordering the ordering of known objects
91  @param unordered Optional: where to put other objects
92 
93  @param unordered_key Optional: the tiebreaker key generator
94  for unordered keys. Default is the original value.
95 
96  """
97  self.order=dict()
98  self.backup_keygen=backup_keygen
99  self.unordered=unordered
100  # self.unordered must NOT cmp() compare to 0 against any
101  # possible index
102  i=0
103  for obj in ordering:
104  while not ( i<self.unordered or i>self.unordered ):
105  i+=1
106  self.order[obj]=i
107  i+=1
108  if self.unordered is None:
109  self.unordered=i
110  ##@var order
111  # Internal ordering information
112  # @protected
113 
114  ##@var backup_keygen
115  # Key generator for tiebreaking
116  # @protected
117 
118  ##@var unordered
119  # Unordered element index
120  # @protected
121 
122  def __call__(self,a):
123  if a in self.order:
124  return (self.order[a],None)
125  elif self.backup_keygen is not None:
126  return (self.unordered,self.backup_keygen(a))
127  return (self.unordered,a)

178 def randint_zeromean(count,imax,randomizer=None):
179  """!Generates "count" numbers uniformly distributed between -imax
180  and imax, inclusive, with a mean of zero.
181  @param count number of numbers to return
182  @param imax maximum value of any number
183  @param randomizer the random module, or something that looks like it"""
184  imax=abs(int(imax))
185  count=int(count)
186  if randomizer is None:
187  randint=random.randint
188  else:
189  randint=randomizer.randint
190  if imax!=0 and not ( -imax < imax):
191  # Integer overflow
192  raise OverflowError(
193  'In randint_zeromean, imax=%d cannot be negated and fit '
194  'within a Python int.'%imax)
195  rand=[ randint(-imax,imax) for x in range(count) ]
196  cen=sum(rand)
197  while cen!=0:
198  if cen>0:
199  while True:
200  icen=randint(0,count-1)
201  if rand[icen]>-imax:
202  rand[icen]-=1
203  break
204  elif cen<0:
205  while True:
206  icen=randint(0,count-1)
207  if rand[icen]<imax:
208  rand[icen]+=1
209  break
210  cen=sum(rand)
211  assert(sum(rand)==0)
212  return rand

252 def timedelta_epsilon(times,rel=None,default=None,sort=False,numerator=10):
253  """!Decides a reasonable epsilon for time equality comparisons.
254 
255  Given an iterable of datetime objects (or anything accepted by
256  to_datetime_rel), computes the minimum time difference between any
257  two adjacent times and divides it by "numerator" (default: 10).
258  The "rel" argument is the relative time when calling
259  to_datetime_rel. If unspecified, it will be the first time seen
260  (in which case that time must be acceptable to to_datetime). The
261  "default" is the return value when all elements in "times" are
262  identical, or when there is only one element. If the default is
263  unspecified and it is needed, then NoTimespan is raised. If sort
264  is specified and True, then the times will be sorted before
265  anything is done (it is False by default).
266 
267  @param times a list of example times for comparison
268  @param rel a reference time to which the times will be compared
269  @param default if too few unique times are found, this is returned
270  If that happens and no default is provided, an exception is raised
271  @param sort if True, sort times
272  @param numerator A measure of how close times should be. The least
273  time difference will be at least numerator times the epsilon
274  @returns an epsilon value"""
275  if sort: times=sorted(times)
276  if rel is not None: rel=to_timedelta(rel)
277  mindiff=None
278  prior=None
279  numerator=int(numerator)
280  for time in times:
281  if rel is None:
282  now=to_datetime(time)
283  rel=now
284  else:
285  now=to_datetime_rel(time,rel)
286  if prior is not None:
287  diff=abs(to_fraction(now-prior,negok=True) / numerator)
288  if mindiff is None or diff<mindiff and diff>0: mindiff=diff
289  prior=now
290  if mindiff is None:
291  if default is None:
292  raise NoTimespan(
293  'Too few non-identical times in input, and no default '
294  'specified. Cannot compute timespan in timedelta_epsilon.',
295  start=rel,end=None)
296  return to_timedelta(default)
297  return to_timedelta(mindiff)

301 def to_fraction(a,b=None,negok=False):
302  """!Converts an object or two to a fraction.
303 
304  This routine is a wrapper around fraction.Fraction() which accepts
305  additional inputs. Fractions are needed to provide higher
306  precision in time and resolution calculations (and also to match
307  the WRF behavior). The arguments are the same as for the
308  fractions.Fraction constructor, but additional calling conventions
309  are accepted: a single float argument, a single datetime.timedelta
310  object, or a string containing an integer and a fraction. If a
311  float is provided, its denominator is restricted to be no more
312  than 1000000. If the resulting fraction is not larger than 0,
313  then InvalidTimestep is thrown unless negok=True.
314 
315  Examples:
316  @code
317  to_fraction(0.855) # 0.855 seconds
318  to_fraction(33) # 33 seconds
319  to_fraction('12/5') # 2.4 seconds
320  to_fraction('7+1/2') # 7.5 seconds
321  to_fraction(0.0000001) # ERROR! Value of the float() is less
322  # than 1e-6
323  to_fraction(1,10000000) # Valid as a fraction, even though value
324  # is less than 1e-6
325  @endcode
326  @param a,b the objects to convert
327  @param negok if True, negative numbers are okay."""
328  result=None
329  if b is not None:
330  result=fractions.Fraction(a,b)
331  elif isinstance(a,datetime.timedelta):
332  result=fractions.Fraction(a.microseconds,1000000)+\
333  a.seconds+24*3600*a.days
334  elif isinstance(a,str): # Catch the 1+3/7 syntax:
335  m=re.match(r'\s*(?P<ipart>[+-]?\d+)\s*(?P<num>[+-]\d+)\s*/\s*(?P<den>\d+)',a)
336  if(m):
337  (i,n,d)=m.groups(['ipart','num','den'])
338  result=fractions.Fraction(int(i)*int(d)+int(n),int(d))
339  elif isinstance(a,float):
340  result=fractions.Fraction.from_float(a).limit_denominator(1000000)
341  if result is None:
342  result=fractions.Fraction(a)
343  if not negok and not result>=0: # avoid < in case of NaN
344  raise InvalidTimestep \
345  ('to_fraction(%s,%s) resulted in a negative timestep' \
346  % (repr(a),repr(b)))
347  return result

351 def to_datetime_rel(d,rel):
352  """!Converts objects to a datetime relative to another datetime.
353 
354  Given a datetime object "rel", converts "d" to a datetime. Object
355  "d" can be anything accepted by to_datetime, or anything accepted
356  as a single argument by to_timedelta. If it is a timedelta, it is
357  added to "rel" to get the final time.
358  @param d the object being converted
359  @param rel the datetime.datetime to which it is to be converted"""
360  if not isinstance(rel,datetime.datetime):
361  rel=to_datetime(rel)
362  if isinstance(d,datetime.datetime):
363  return d
364  elif isinstance(d,datetime.timedelta):
365  return rel+d
366  elif isinstance(d,str):
367  if(re.match(r'\A(?:\d\d\d\d-\d\d-\d\d \d\d:\d\d:\d\d|\d{10}|\d{12})\Z',d)):
368  if len(d)==10:
369  return datetime.datetime.strptime(d,'%Y%m%d%H')
370  elif len(d)==12:
371  return datetime.datetime.strptime(d,'%Y%m%d%H%M')
372  else:
373  return datetime.datetime.strptime(d,'%Y-%m-%d %H:%M:%S')
374  return rel+to_timedelta(d)

403 def to_timedelta(a,b=None,negok=True):
404  """!Converts an object to a datetime.timedelta.
405 
406  Returns a datetime.timedelta object. If "a" is a timedelta, then
407  that value is returned. If "a" is a string of the format 08:13 or
408  08:13:12, optionally preceded by a "-" then it is interpreted as
409  HH:MM or HH:MM:SS, respectively (where a "-" negates). Otherwise,
410  the arguments are sent to the to_fraction() function, and the
411  result is converted into a timedelta.
412 
413  @param a object being converted
414  @param b second object, if a time range is sent
415  @param negok if True, negative timespans are allowed
416  @returns a datetime.timedelta"""
417  if isinstance(a,datetime.timedelta): return a
418  if isinstance(a,str) and b is None:
419  # 03:14 = three hours
420  try:
421  m=re.search(r'''(?ix) \A \s* (?P<negative>-)? 0* (?P<hours>\d+)
422  :0*(?P<minutes>\d+)
423  (?: :0*(?P<seconds>\d+(?:\.\d*)?) )?
424  \s*''', a)
425  if m:
426  (hours,minutes,seconds)=(0.,0.,0.)
427  mdict=m.groupdict()
428  if 'hours' in mdict and mdict['hours'] is not None:
429  hours=float(mdict['hours'])
430  if 'minutes' in mdict and mdict['minutes'] is not None:
431  minutes=float(mdict['minutes'])
432  if 'seconds' in mdict and mdict['seconds'] is not None:
433  seconds=float(mdict['seconds'])
434  dt=datetime.timedelta(hours=hours,minutes=minutes,
435  seconds=seconds)
436  if 'negative' in mdict and mdict['negative'] is not None \
437  and mdict['negative']=='-':
438  return -dt
439  return dt
440  except(TypeError,ValueError,AttributeError) as e:
441  # Fall back to to_fraction
442  raise #FIXME: remove this line? Why is this here? It
443  #conflicts with the description.
444  pass
445  f=to_fraction(a,b,negok=negok)
446  (fpart,ipart)=math.modf(f)
447  return datetime.timedelta(seconds=ipart,
448  microseconds=int(round(fpart*1e6)))

525 class TimeContainer(object):
526  """!Abstract base class that maps from time to objects.
527 
528  This is the abstract base class of any class that represents a
529  mapping from a set of times to a set of data. It provides the
530  underlying implementation of TimeArray and TimeMapping. This is
531  not exported by "from produtil.numerics import *" to prevent
532  accidental use."""
533  # Implementation notes:
534  def __init__(self,times,init=None):
535  """!TimeContainer constructor.
536 
537  Initializes the internal arrays for the given list of times,
538  which must not be empty. Note that strange, potentially bad
539  things will happen if there are duplicate times.
540 
541  Implementation notes:
542 
543  @code
544  self._times[N]: a list of times
545  self._data[N]: the data for each time
546  self._assigned[N]: True if there is data for this time,
547  False otherwise
548  @endcode
549 
550  where N is the length of the input times array. The _data
551  will be filled with init() and _assigned filled with True if
552  init is present and not None, otherwise _assigned will be
553  False.
554 
555  @param times the times to map from
556  @param init a constructor or function that creates initial
557  values for all uninitialized array elements"""
558  self._times=times
559  N=len(self._times)
560  if init is None:
561  self._assigned=[False]*N
562  self._data=[None]*N
563  else:
564  self._assigned=[True]*N
565  self._data=[init() for x in range(N)]
566  def at_index(self,index):
567  """!Returns the data at the given index, or raises KeyError if
568  no data exists.
569  @param index the desired index"""
570  if not self._assigned[index]: raise KeyError(index)
571  return self._data[index]
572  def index_of(self,when):
573  """!This virtual function should be implemented in a subclass.
574  It should return the index of the time to use for the given
575  time.
576  @param when the time of interest"""
577  raise NotImplementedError(
578  'TimeContainer.index_of is not implemented.')
579  @property
580  def lasttime(self):
581  """!Returns the last time in the array or list of times, even
582  if it has no data."""
583  return self._times[-1]
584  @property
585  def firsttime(self):
586  """!Returns the first time in the array or list of times, even
587  if it has no data."""
588  return self._times[0]

590  """!Returns the item at the latest time that is not later than
591  "when". If there is no data assigned at that time, then
592  KeyError is raised.
593  @param when the time of interest"""
594  (iwhen,index)=self.index_of(when)
595  if not self._assigned[index]:
596  raise KeyError(when)
597  return self._data[index]
598  def neartime(self,when,epsilon=None):
599  """!Returns a tuple containing the time nearest to "when"
600  without going over, and the index of that time.
601 
602  @param epsilon If specified, raise
603  hwrf.exceptions.NoNearbyValues if no times are near that
604  value.
605  @param when the time of interest"""
606  (then,index)=self.index_of(when)
607  if epsilon is not None:
608  if abs(to_fraction(then-when,negok=True)) \
609  <= to_fraction(epsilon):
610  return then
611  else:
612  raise NoNearbyValues(
613  '%s: nearest value not after is %s, which is not '
614  'within %s seconds.'%(str(when),str(then),

616  else:
617  return then
618  def get(self,when,default):
619  """!Returns the item at the latest time that is not later than

791  times do not have to be exactly at an interval.

794  timestep (the greatest common factor) is used since most of the
795  TimeContainer methods only work on the indices that have data.
