224  def get_all_field_info(self, var_list, data_type):
225  """!Get field info based on data type"""
226  field_list = []
227  for var_info in var_list:
228  if data_type == 'FCST':
229  level = var_info['fcst_level']
230  thresh = var_info['fcst_thresh']
231  name = var_info['fcst_name']
232  extra = var_info['fcst_extra']
233  elif data_type == 'OBS':
234  level = var_info['obs_level']
235  thresh = var_info['obs_thresh']
236  name = var_info['obs_name']
237  extra = var_info['obs_extra']
238  elif data_type == 'ENS':
239  level = var_info['ens_level']
240  thresh = var_info['ens_thresh']
241  name = var_info['ens_name']
242  extra = var_info['ens_extra']
243  else:
244  return ''
245 
246  # check if python embedding is used and set up correctly
247  # set env var for file type if it is used
248  pyEmbedIsOK = self.check_for_python_embedding(data_type, var_info)
249  if not pyEmbedIsOK:
250  return ''
251 
252  next_field = self.get_field_info(v_level=level,
253  v_thresh=thresh,
254  v_name=name,
255  v_extra=extra,
256  d_type=data_type)
257  if next_field is None:
258  return ''
259 
260  field_list.extend(next_field)
261 
262  return ','.join(field_list)

265  def find_model_members(self, time_info):
266  """! Finds the model member files to compare
267  Args:
268  @param time_info dictionary containing timing information
269  @rtype string
270  @return Returns a list of the paths to the ensemble model files
271  """
272  model_dir = self.c_dict['FCST_INPUT_DIR']
273  # used for filling in missing files to ensure ens_thresh check is accurate
274  fake_dir = '/ensemble/member/is/missing'
275 
276  # model_template is a list of 1 or more.
277  ens_members_path = []
278 
279  # get all files that exist
280  for ens_member_template in self.c_dict['FCST_INPUT_TEMPLATE']:
281  model_ss = StringSub(self.logger,
282  ens_member_template,
283  **time_info)
284  member_file = model_ss.do_string_sub()
285  expected_path = os.path.join(model_dir, member_file)
286 
287  # if wildcard expression, get all files that match
288  if '?' in expected_path:
289  wildcard_files = sorted(glob.glob(expected_path))
290  self.logger.debug('Ensemble members file pattern: {}'
291  .format(expected_path))
292  self.logger.debug('{} members match file pattern'
293  .format(str(len(wildcard_files))))
294 
295  # add files to list of ensemble members
296  for wildcard_file in wildcard_files:
297  ens_members_path.append(wildcard_file)
298  else:
299  # otherwise check if file exists
300  expected_path = util.preprocess_file(expected_path,
301  self.c_dict['FCST_INPUT_DATATYPE'],
302  self.config)
303 
304  # if the file exists, add it to the list
305  if expected_path:
306  ens_members_path.append(expected_path)
307  else:
308  # add relative path to fake dir and add to list
309  ens_members_path.append(os.path.join(fake_dir, member_file))
310  self.logger.warning('Expected ensemble file {} not found'
311  .format(member_file))
312 
313  # if more files found than expected, error and exit
314  if len(ens_members_path) > self.c_dict['N_MEMBERS']:
315  msg = 'Found more files than expected! ' +\
316  'Found {} expected {}. '.format(len(ens_members_path),
317  self.c_dict['N_MEMBERS']) +\
318  'Adjust wildcard expression in [filename_templates] '+\
319  'FCST_ENSEMBLE_STAT_INPUT_TEMPLATE or adjust [config] '+\
320  'ENSEMBLE_STAT_N_MEMBERS. Files found: {}'.format(ens_members_path)
321  self.log_error(msg)
322  self.logger.error("Could not file files in {} for init {} f{} "
323  .format(model_dir, time_info['init_fmt'],
324  str(time_info['lead_hours'])))
325  return False
326  # if fewer files found than expected, warn and add fake files
327  elif len(ens_members_path) < self.c_dict['N_MEMBERS']:
328  msg = 'Found fewer files than expected. '+\
329  'Found {} expected {}.'.format(len(ens_members_path),
330  self.c_dict['N_MEMBERS'])
331  self.logger.warning(msg)
332  # add fake files to list to get correct number of files for ens_thresh
333  diff = self.c_dict['N_MEMBERS'] - len(ens_members_path)
334  self.logger.warning('Adding {} fake files to '.format(str(diff))+\
335  'ensure ens_thresh check is accurate')
336  for _ in range(0, diff, 1):
337  ens_members_path.append(fake_dir)
338 
339  # write file that contains list of ensemble files
340  list_filename = time_info['init_fmt'] + '_' + \
341  str(time_info['lead_hours']) + '_ensemble.txt'
342  return self.write_list_file(list_filename, ens_members_path)
